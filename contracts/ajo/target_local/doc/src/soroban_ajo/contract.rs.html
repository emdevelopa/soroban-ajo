58    /// Create a new Ajo group.
59    ///
60    /// Initializes a new rotating savings group with the specified parameters.
61    /// The creator becomes the first member and the contract validates all parameters
62    /// before storage. A unique group ID is assigned and returned.
63    ///
64    /// # Arguments
65    /// * `env` - The Soroban contract environment
66    /// * `creator` - Address of the group creator (automatically becomes first member)
67    /// * `contribution_amount` - Fixed amount each member contributes per cycle (in stroops, must be > 0)
68    /// * `cycle_duration` - Duration of each cycle in seconds (must be > 0)
69    /// * `max_members` - Maximum number of members allowed in the group (must be >= 2 and <= 100)
70    ///
71    /// # Returns
72    /// The unique group ID assigned to the new group
73    ///
74    /// # Errors
75    /// * `ContributionAmountZero` - If contribution_amount == 0
76    /// * `ContributionAmountNegative` - If contribution_amount < 0
77    /// * `CycleDurationZero` - If cycle_duration == 0
78    /// * `MaxMembersBelowMinimum` - If max_members < 2
79    /// * `MaxMembersAboveLimit` - If max_members > 100
80    pub fn create_group(
81        env: Env,
82        creator: Address,
83        contribution_amount: i128,
84        cycle_duration: u64,
85        max_members: u32,
86    ) -> Result<u64, AjoError> {
87        // Validate parameters
88        utils::validate_group_params(contribution_amount, cycle_duration, max_members)?;
89
90        // Require authentication
91        creator.require_auth();
92
93        // Generate new group ID
94        let group_id = storage::get_next_group_id(&env);
95
96        // Initialize members list with creator
97        let mut members = Vec::new(&env);
98        members.push_back(creator.clone());
99
100        // Get current timestamp
101        let now = utils::get_current_timestamp(&env);
102
103        // Create group
104        let group = Group {
105            id: group_id,
106            creator: creator.clone(),
107            contribution_amount,
108            cycle_duration,
109            max_members,
110            members,
111            current_cycle: 1,
112            payout_index: 0,
113            created_at: now,
114            cycle_start_time: now,
115            is_complete: false,
116        };
117
118        // Store group
119        storage::store_group(&env, group_id, &group);
120
121        // Emit event
122        events::emit_group_created(&env, group_id, &creator, contribution_amount, max_members);
123
124        Ok(group_id)
125    }

164    /// Join an existing group.
165    ///
166    /// Adds a new member to an active group if space is available.
167    /// The member's authentication is required. The member cannot join if they
168    /// are already a member, the group is full, or the group has completed all cycles.
169    ///
170    /// # Arguments
171    /// * `env` - The Soroban contract environment
172    /// * `member` - Address of the member joining (must authenticate)
173    /// * `group_id` - The group to join
174    ///
175    /// # Returns
176    /// `Ok(())` on successful group join
177    ///
178    /// # Errors
179    /// * `GroupNotFound` - If the group does not exist
180    /// * `MaxMembersExceeded` - If the group has reached max members
181    /// * `AlreadyMember` - If the address is already a member
182    /// * `GroupComplete` - If the group has completed all cycles
183    pub fn join_group(env: Env, member: Address, group_id: u64) -> Result<(), AjoError> {
184        // Require authentication
185        member.require_auth();
186
187        // Get group
188        let mut group = storage::get_group(&env, group_id).ok_or(AjoError::GroupNotFound)?;
189
190        // Check if group is complete
191        if group.is_complete {
192            return Err(AjoError::GroupComplete);
193        }
194
195        // Check if already a member
196        if utils::is_member(&group.members, &member) {
197            return Err(AjoError::AlreadyMember);
198        }
199
200        // Check if group is full
201        if group.members.len() >= group.max_members {
202            return Err(AjoError::MaxMembersExceeded);
203        }
204
205        // Add member
206        group.members.push_back(member.clone());
207
208        // Update storage
209        storage::store_group(&env, group_id, &group);
210
211        // Emit event
212        events::emit_member_joined(&env, group_id, &member);
213
214        Ok(())
215    }

237    ///
238    /// Records a member's contribution for the current cycle. Each member can contribute
239    /// once per cycle. Authentication is required. Contributions are recorded but actual
240    /// fund transfers are handled by external payment systems.
241    ///
242    /// # Arguments
243    /// * `env` - The Soroban contract environment
244    /// * `member` - Address making the contribution (must authenticate)
245    /// * `group_id` - The group to contribute to
246    ///
247    /// # Returns
248    /// `Ok(())` on successful contribution recording
249    ///
250    /// # Errors
251    /// * `GroupNotFound` - If the group does not exist
252    /// * `NotMember` - If the address is not a member
253    /// * `AlreadyContributed` - If already contributed this cycle
254    /// * `GroupComplete` - If the group has completed all cycles
255    pub fn contribute(env: Env, member: Address, group_id: u64) -> Result<(), AjoError> {
256        // Require authentication
257        member.require_auth();
258
259        // Get group
260        let group = storage::get_group(&env, group_id).ok_or(AjoError::GroupNotFound)?;
261
262        // Check if group is complete
263        if group.is_complete {
264            return Err(AjoError::GroupComplete);
265        }
266
267        // Check if member
268        if !utils::is_member(&group.members, &member) {
269            return Err(AjoError::NotMember);
270        }
271
272        // Check if already contributed
273        if storage::has_contributed(&env, group_id, group.current_cycle, &member) {
274            return Err(AjoError::AlreadyContributed);
275        }
276
277        // Transfer contribution to contract
278        // Note: In production, this would use token.transfer() or native transfer
279        // For now, we mark as contributed (assuming payment succeeded)
280
281        // Record contribution
282        storage::store_contribution(&env, group_id, group.current_cycle, &member, true);
283
284        // Emit event
285        events::emit_contribution_made(
286            &env,
287            group_id,
288            &member,
289            group.current_cycle,
290            group.contribution_amount,
291        );
292
293        Ok(())
294    }

325    /// Execute payout for the current cycle.
326    ///
327    /// This is the core function that rotates payouts through group members.
328    /// It verifies that all members have contributed, calculates the total payout,
329    /// distributes funds to the next recipient, and advances the cycle.
330    /// When all members have received their payout, the group is marked complete.
331    ///
332    /// Process:
333    /// 1. Verifies all members have contributed in the current cycle
334    /// 2. Calculates total payout (contribution_amount Ã— member_count)
335    /// 3. Records payout to the current recipient
336    /// 4. Emits payout event
337    /// 5. Advances to next cycle (or marks complete if done)
338    ///
339    /// # Arguments
340    /// * `env` - The Soroban contract environment
341    /// * `group_id` - The group to execute payout for
342    ///
343    /// # Returns
344    /// `Ok(())` on successful payout execution
345    ///
346    /// # Errors
347    /// * `GroupNotFound` - If the group does not exist
348    /// * `IncompleteContributions` - If not all members have contributed
349    /// * `GroupComplete` - If the group has already completed all payouts
350    /// * `NoMembers` - If the group has no members (should never happen)
351    pub fn execute_payout(env: Env, group_id: u64) -> Result<(), AjoError> {
352        // Get group
353        let mut group = storage::get_group(&env, group_id).ok_or(AjoError::GroupNotFound)?;
354
355        // Check if group is complete
356        if group.is_complete {
357            return Err(AjoError::GroupComplete);
358        }
359
360        // Check if all members have contributed
361        if !utils::all_members_contributed(&env, &group) {
362            return Err(AjoError::IncompleteContributions);
363        }
364
365        // Get payout recipient
366        let payout_recipient = group
367            .members
368            .get(group.payout_index)
369            .ok_or(AjoError::NoMembers)?;
370
371        // Calculate payout amount
372        let payout_amount = utils::calculate_payout_amount(&group);
373
374        // Transfer payout to recipient
375        // Note: In production, this would use token.transfer() or native transfer
376        // For now, we just record it
377
378        // Mark payout as received
379        storage::mark_payout_received(&env, group_id, &payout_recipient);
380
381        // Emit payout event
382        events::emit_payout_executed(
383            &env,
384            group_id,
385            &payout_recipient,
386            group.current_cycle,
387            payout_amount,
388        );
389
390        // Advance payout index
391        group.payout_index += 1;
392
393        // Check if all members have received payout
394        if group.payout_index >= group.members.len() {
395            // All members have received payout - mark complete
396            group.is_complete = true;
397            events::emit_group_completed(&env, group_id);
398        } else {
399            // Advance to next cycle
400            group.current_cycle += 1;
401            group.cycle_start_time = utils::get_current_timestamp(&env);
402        }
403
404        // Update storage
405        storage::store_group(&env, group_id, &group);
406
407        Ok(())
408    }

429    /// Get comprehensive group status.
430    ///
431    /// Returns a detailed snapshot of the group's current state, including cycle
432    /// information, contribution status, payout progress, and timing data.
433    /// This is the primary function for checking a group's overall progress.
434    ///
435    /// Returns information about:
436    /// - Current cycle number and progress
437    /// - Next recipient for payout
438    /// - Members who have contributed and those who are pending
439    /// - Cycle timing (start time, end time, whether cycle is active)
440    /// - Whether the group is complete
441    ///
442    /// # Arguments
443    /// * `env` - The Soroban contract environment
444    /// * `group_id` - The unique group identifier
445    ///
446    /// # Returns
447    /// A `GroupStatus` struct containing comprehensive group state information
448    ///
449    /// # Errors
450    /// * `GroupNotFound` - If the group does not exist
451    pub fn get_group_status(env: Env, group_id: u64) -> Result<GroupStatus, AjoError> {
452        // Get the group data
453        let group = storage::get_group(&env, group_id).ok_or(AjoError::GroupNotFound)?;
454
455        // Get current timestamp
456        let current_time = utils::get_current_timestamp(&env);
457
458        // Calculate cycle timing
459        let cycle_end_time = group.cycle_start_time + group.cycle_duration;
460        let is_cycle_active = current_time < cycle_end_time;
461
462        // Get contribution status for all members in current cycle
463        let contributions =
464            storage::get_cycle_contributions(&env, group_id, group.current_cycle, &group.members);
465
466        // Count contributions and build pending list
467        let mut contributions_received: u32 = 0;
468        let mut pending_contributors = Vec::new(&env);
469
470        for (member, has_contributed) in contributions.iter() {
471            if has_contributed {
472                contributions_received += 1;
473            } else {
474                pending_contributors.push_back(member);
475            }
476        }
477
478        // Determine next recipient
479        let (has_next_recipient, next_recipient) = if group.is_complete {
480            // Use placeholder (creator) when complete
481            (false, group.creator.clone())
482        } else {
483            // Get the member at payout_index
484            let recipient = group
485                .members
486                .get(group.payout_index)
487                .unwrap_or_else(|| group.creator.clone());
488            (true, recipient)
489        };
490
491        // Build and return status
492        Ok(GroupStatus {
493            group_id: group.id,
494            current_cycle: group.current_cycle,
495            has_next_recipient,
496            next_recipient,
497            contributions_received,
498            total_members: group.members.len() as u32,
499            pending_contributors,
500            is_complete: group.is_complete,
501            is_cycle_active,
502            cycle_start_time: group.cycle_start_time,
503            cycle_end_time,
504            current_time,
505        })
506    }
