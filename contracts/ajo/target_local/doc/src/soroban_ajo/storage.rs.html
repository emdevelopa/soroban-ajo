3/// Logical storage key categories used by the Ajo contract.
4///
5/// Soroban storage uses raw key values; this enum documents the naming
6/// conventions and composite key structures used throughout the contract.
7/// Each variant maps to a specific `symbol_short!` or tuple key at the
8/// call site.
9pub enum StorageKey {
10    /// Singleton key for the contract administrator address.
11    /// Stored in instance storage under `"ADMIN"`.
12    Admin,
13
14    /// Monotonically increasing counter used to assign unique group IDs.
15    /// Stored in instance storage under `"GCOUNTER"`.
16    GroupCounter,
17
18    /// Full group state keyed by its numeric ID.
19    /// Stored in persistent storage under `("GROUP", group_id)`.
20    Group(u64),
21
22    /// Per-member contribution flag for a specific group and cycle.
23    /// Stored in persistent storage under `("CONTRIB", group_id, cycle, member)`.
24    /// Value is `bool` — `true` means the member has contributed.
25    Contribution(u64, u32, Address),
26
27    /// Records whether a member has received their payout for a group.
28    /// Stored in persistent storage under `("PAYOUT", group_id, member)`.
29    /// Value is `bool` — `true` means the payout has been distributed.
30    PayoutReceived(u64, Address),
31}
32
33impl StorageKey {
34    /// Returns the short [`Symbol`] prefix associated with this key variant.
35    ///
36    /// This method extracts the symbol portion of a storage key. Note that composite
37    /// keys (e.g., [`StorageKey::Group`]) pair this symbol with additional fields
38    /// in a tuple at the storage call site; this method returns only the symbol portion.
39    ///
40    /// # Arguments
41    /// * `env` - The contract environment (needed for symbol creation)
42    ///
43    /// # Returns
44    /// The [`Symbol`] corresponding to this key variant's prefix
45    pub fn to_symbol(&self, env: &Env) -> Symbol {
46        match self {
47            StorageKey::Admin => symbol_short!("ADMIN"),
48            StorageKey::GroupCounter => symbol_short!("GCOUNTER"),
49            StorageKey::Group(_) => symbol_short!("GROUP"),
50            StorageKey::Contribution(_, _, _) => symbol_short!("CONTRIB"),
51            StorageKey::PayoutReceived(_, _) => symbol_short!("PAYOUT"),
52        }
53    }
54}

56/// Returns the next available group ID and atomically increments the counter.
57///
58/// The counter starts at 0 and is stored in instance storage. Each call
59/// increments it by 1 and returns the new value, so the first group
60/// ever created receives ID `1`.
61///
62/// # Arguments
63/// * `env` - The contract environment used to access instance storage
64///
65/// # Returns
66/// The next available group ID (starting from 1)
67pub fn get_next_group_id(env: &Env) -> u64 {
68    let key = symbol_short!("GCOUNTER");
69    let current: u64 = env.storage().instance().get(&key).unwrap_or(0);
70    let next = current + 1;
71    env.storage().instance().set(&key, &next);
72    next
73}

75/// Persists a [`Group`](crate::types::Group) to persistent ledger storage.
76///
77/// This function writes or overwrites the group data for the given `group_id`.
78/// Call this any time the group's mutable fields (members, cycle, payout index, etc.) change.
79///
80/// # Arguments
81/// * `env` - The contract environment used to access persistent storage
82/// * `group_id` - The unique identifier for the group
83/// * `group` - The group data to store
84pub fn store_group(env: &Env, group_id: u64, group: &crate::types::Group) {
85    let key = (symbol_short!("GROUP"), group_id);
86    env.storage().persistent().set(&key, group);
87}

89/// Retrieves a [`Group`](crate::types::Group) from persistent ledger storage.

101pub fn get_group(env: &Env, group_id: u64) -> Option<crate::types::Group> {

104}

120/// Records whether a member has paid their contribution for a given cycle.
121///
122/// The composite key `(group_id, cycle, member)` ensures that contribution
123/// records are scoped per-group and per-cycle. A member's contribution in
124/// cycle 1 does not affect their status in cycle 2.
125///
126/// # Arguments
127/// * `env` - The contract environment used to access persistent storage
128/// * `group_id` - The group the contribution belongs to
129/// * `cycle` - The cycle number being recorded
130/// * `member` - The contributing member's address
131/// * `paid` - `true` to mark as paid; `false` to reset (rarely needed)
132pub fn store_contribution(env: &Env, group_id: u64, cycle: u32, member: &Address, paid: bool) {
133    let key = (symbol_short!("CONTRIB"), group_id, cycle, member);
134    env.storage().persistent().set(&key, &paid);
135}

137/// Returns `true` if the given member has contributed during the specified cycle.
138///
139/// Defaults to `false` if no record exists, meaning the member has not yet contributed.
140///
141/// # Arguments
142/// * `env` - The contract environment used to access persistent storage
143/// * `group_id` - The group to check
144/// * `cycle` - The cycle number to check
145/// * `member` - The member address to check
146///
147/// # Returns
148/// `true` if the member has contributed, `false` otherwise
149pub fn has_contributed(env: &Env, group_id: u64, cycle: u32, member: &Address) -> bool {
150    let key = (symbol_short!("CONTRIB"), group_id, cycle, member);

152}

184/// Returns contribution status for every member in a cycle as an ordered vector.
185///
186/// Iterates through `members` in order and looks up each one's contribution
187/// flag for the given cycle. The returned vector preserves member order and
188/// pairs each address with a `bool` indicating whether they have contributed.
189///
190/// # Arguments
191/// * `env` - The contract environment used to access persistent storage
192/// * `group_id` - The group to query
193/// * `cycle` - The cycle number to query
194/// * `members` - The ordered member list from the group (use `group.members`)
195///
196/// # Returns
197/// A `Vec<(Address, bool)>` with members in original order, `true` = has contributed
198pub fn get_cycle_contributions(
199    env: &Env,
200    group_id: u64,
201    cycle: u32,
202    members: &Vec<Address>,
203) -> Vec<(Address, bool)> {
204    let mut results = Vec::new(env);
205    for member in members.iter() {
206        let paid = has_contributed(env, group_id, cycle, &member);
207        results.push_back((member, paid));
208    }
209    results
210}
