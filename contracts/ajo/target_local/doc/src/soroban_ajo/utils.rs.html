5/// Returns `true` if `address` appears in the group's `members` list.
6///
7/// Performs a linear scan since Soroban's `Vec` does not support O(1) lookup.
8/// For groups capped at 100 members this is acceptable.
9///
10/// # Arguments
11/// * `members` - The ordered member list to search
12/// * `address` - The address to look for
13///
14/// # Returns
15/// `true` if the address is found in the members list, `false` otherwise
16pub fn is_member(members: &Vec<Address>, address: &Address) -> bool {
17    for member in members.iter() {
18        if member == *address {
19            return true;
20        }
21    }
22    false
23}

25/// Returns `true` if every member of the group has contributed in the current cycle.
26///
27/// Iterates over all members and short-circuits on the first missing contribution.
28/// This is called by [`execute_payout`](crate::contract::AjoContract::execute_payout)
29/// to gate payout execution — a payout cannot proceed until this returns `true`.
30///
31/// # Arguments
32/// * `env` - The contract environment (needed for storage reads)
33/// * `group` - The group whose current cycle contributions are being verified
34///
35/// # Returns
36/// `true` if all members have contributed, `false` otherwise
37pub fn all_members_contributed(env: &Env, group: &Group) -> bool {
38    for member in group.members.iter() {
39        if !crate::storage::has_contributed(env, group.id, group.current_cycle, &member) {
40            return false;
41        }
42    }
43    true
44}

46/// Calculates the total payout amount for a single cycle.
47///
48/// The payout equals each member's fixed contribution multiplied by the total
49/// number of members. This ensures the recipient receives the full pool of contributions.
50///
51/// # Arguments
52/// * `group` - The group whose payout is being calculated
53///
54/// # Returns
55/// Total payout in stroops (`contribution_amount × member_count`)
56pub fn calculate_payout_amount(group: &Group) -> i128 {
57    let member_count = group.members.len() as i128;
58    group.contribution_amount * member_count
59}

135/// Returns the start and end timestamps for the group's current cycle window.
136///
137/// The cycle window is `[cycle_start_time, cycle_start_time + cycle_duration)`.
138/// This is a pure calculation that does not read the ledger clock.
139///
140/// # Arguments
141/// * `group` - The group whose cycle window is being computed
142/// * `_current_time` - Reserved for future use (e.g., dynamic window adjustment)
143///
144/// # Returns
145/// A `(start, end)` tuple of Unix timestamps in seconds
146pub fn get_cycle_window(group: &Group, _current_time: u64) -> (u64, u64) {
147    let cycle_start = group.cycle_start_time;
148    let cycle_end = cycle_start + group.cycle_duration;
149    (cycle_start, cycle_end)
150}

152/// Returns `true` if `current_time` falls within the group's active cycle window.
153///
154/// The window is inclusive of `cycle_start_time` and exclusive of `cycle_end_time`
155/// (`start <= current_time < end`). Once this returns `false`, the cycle has expired
156/// and a payout can be triggered.
157///
158/// # Arguments
159/// * `group` - The group whose cycle window is being checked
160/// * `current_time` - The timestamp to evaluate, typically from [`get_current_timestamp`]
161///
162/// # Returns
163/// `true` if current_time is within the cycle window, `false` otherwise
164pub fn is_within_cycle_window(group: &Group, current_time: u64) -> bool {
165    let (cycle_start, cycle_end) = get_cycle_window(group, current_time);
166    current_time >= cycle_start && current_time < cycle_end
