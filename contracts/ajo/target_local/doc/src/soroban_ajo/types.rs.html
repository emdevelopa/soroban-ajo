3/// Represents an Ajo group configuration and state.
4///
5/// An Ajo (also known as Esusu or Tontine) is a rotating savings group
6/// where members contribute a fixed amount each cycle, and one member
7/// receives the full pool each round until everyone has been paid out.
8#[contracttype]
9#[derive(Clone, Debug, Eq, PartialEq)]
10pub struct Group {
11    /// Unique group identifier, auto-incremented from storage counter
12    pub id: u64,
13
14    /// Address of the member who created the group.
15    /// Automatically added as the first member on creation.
16    pub creator: Address,
17
18    /// Fixed contribution amount each member must pay per cycle, denominated in stroops.
19    /// 1 XLM = 10,000,000 stroops.
20    pub contribution_amount: i128,
21
22    /// Duration of each cycle in seconds.
23    /// When a cycle ends, the next payout can be triggered.
24    pub cycle_duration: u64,
25
26    /// Maximum number of members allowed in the group.
27    /// Must be between 2 and 100 (inclusive).
28    pub max_members: u32,
29
30    /// Ordered list of member addresses.
31    /// Members receive payouts in the order they appear in this list.
32    pub members: Vec<Address>,
33
34    /// Current cycle number, starts at 1 and increments after each payout.
35    pub current_cycle: u32,
36
37    /// Zero-based index into `members` indicating who receives the next payout.
38    /// When `payout_index == members.len()`, the group is complete.
39    pub payout_index: u32,
40
41    /// Unix timestamp (seconds) when the group was created.
42    pub created_at: u64,
43
44    /// Unix timestamp (seconds) when the current cycle started.
45    /// Used together with `cycle_duration` to calculate when the cycle ends.
46    pub cycle_start_time: u64,
47
48    /// Whether the group has completed all payout cycles.
49    /// Once `true`, no further contributions or payouts are accepted.
50    pub is_complete: bool,
51}

53/// Records a single member's contribution for a specific cycle.
54///
55/// Written to persistent storage when a member calls `contribute()`.
56/// Used to prevent double-contributions and to verify cycle completion
57/// before executing a payout.
58#[contracttype]
59#[derive(Clone, Debug, Eq, PartialEq)]
60pub struct ContributionRecord {
61    /// Address of the member who made (or is expected to make) the contribution.
62    pub member: Address,
63
64    /// The group this contribution belongs to.
65    pub group_id: u64,
66
67    /// The cycle number this contribution is for.
68    pub cycle: u32,
69
70    /// Whether the member has paid their contribution for this cycle.
71    /// `true` means the contribution has been recorded; `false` means pending.
72    pub has_paid: bool,
73
74    /// Unix timestamp (seconds) when the contribution was recorded.
75    pub timestamp: u64,
76}
77
78/// Records that a member has received their payout for a given cycle.
79///
80/// Written to persistent storage after `execute_payout()` succeeds.
81/// Can be used for audit trails and to prevent duplicate payouts.
82#[contracttype]
83#[derive(Clone, Debug, Eq, PartialEq)]
84pub struct PayoutRecord {
85    /// Address of the member who received the payout.
86    pub member: Address,
87
88    /// The group this payout belongs to.
89    pub group_id: u64,
90
91    /// The cycle number in which this payout was made.
92    pub cycle: u32,
93
94    /// Total payout amount in stroops (`contribution_amount Ã— member_count`).
95    pub amount: i128,
96
97    /// Unix timestamp (seconds) when the payout was executed.
98    pub timestamp: u64,
99}

101/// Comprehensive snapshot of a group's current state.
102///
103/// Returned by [`AjoContract::get_group_status`] to give callers a single
104/// consolidated view without having to make multiple queries.
105#[contracttype]
106#[derive(Clone, Debug, Eq, PartialEq)]
107pub struct GroupStatus {
108    /// The unique identifier of the group being described.
109    pub group_id: u64,
110
111    /// Current cycle number (1-based). Increments after each successful payout.
112    pub current_cycle: u32,
113
114    /// `true` if there is a valid next recipient (i.e., the group is not yet complete).
115    /// When `false`, `next_recipient` is a placeholder and should be ignored.
116    pub has_next_recipient: bool,
117
118    /// Address of the member scheduled to receive the next payout.
119    /// Only meaningful when `has_next_recipient` is `true`.
120    pub next_recipient: Address,
121
122    /// Number of members who have already contributed in the current cycle.
123    pub contributions_received: u32,
124
125    /// Total number of members currently in the group.
126    pub total_members: u32,
127
128    /// Addresses of members who have not yet contributed in the current cycle.
129    pub pending_contributors: Vec<Address>,
130
131    /// Whether the group has finished all cycles and is closed.
132    pub is_complete: bool,
133
134    /// Whether the current cycle window is still open for contributions.
135    /// `false` means the cycle has expired and a payout can be triggered.
136    pub is_cycle_active: bool,
137
138    /// Unix timestamp (seconds) when the current cycle started.
139    pub cycle_start_time: u64,
140
141    /// Unix timestamp (seconds) when the current cycle ends (`cycle_start_time + cycle_duration`).
142    pub cycle_end_time: u64,
143
144    /// The ledger timestamp at the moment this status was queried.
145    pub current_time: u64,
146}
